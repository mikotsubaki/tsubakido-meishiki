function getS(y){ return ST[y]||ST[2000]; }

function dateUTC(y,m,d,h,mi){
  return Date.UTC(y,m-1,d,h||0,mi||0,0);
}

function getSetsuDateUTC(y,m){
  var s=getS(y);
  return dateUTC(y,m,s[m-1],0,0);
}

function getPrevSetsuUTC(y,m,d,h,mi){
  var birth=dateUTC(y,m,d,h,mi);
  var cm=m, cy=y, cur=getSetsuDateUTC(cy,cm);
  if(birth>=cur) return cur;
  cm--;
  if(cm<=0){cm=12;cy--;}
  return getSetsuDateUTC(cy,cm);
}

function getNextSetsuUTC(y,m,d,h,mi){
  var birth=dateUTC(y,m,d,h,mi);
  var cm=m, cy=y, cur=getSetsuDateUTC(cy,cm);
  if(birth<cur) return cur;
  cm++;
  if(cm>12){cm=1;cy++;}
  return getSetsuDateUTC(cy,cm);
}

function getYear(y,m,d,h,mi){
  var yr=y;
  var risshun=getSetsuDateUTC(y,2);
  if(dateUTC(y,m,d,h,mi)<risshun) yr=y-1;
  var idx=((yr-4)%60+60)%60;
  return {k:idx%10,s:idx%12};
}

function getMonth(y,m,d,h,mi){
  var mo=m;
  if(dateUTC(y,m,d,h,mi)<getSetsuDateUTC(y,m)){
    mo=m-1;
    if(mo<=0) mo=12;
  }
  var yc=getYear(y,m,d,h,mi);
  var shi=MSH[mo-1];
  var kan=(MKB[yc.k]+SFT.indexOf(shi))%10;
  return {k:kan,s:shi};
}

function getDay(y,m,d,h,mi){
  // 0時切替が初期設定。23時切替流派にしたい場合はCONFIG.dayBoundaryHour=23。
  var adjY=y, adjM=m, adjD=d;
  if(CONFIG.dayBoundaryHour===23 && h>=23){
    var nd=new Date(Date.UTC(y,m-1,d)+86400000);
    adjY=nd.getUTCFullYear(); adjM=nd.getUTCMonth()+1; adjD=nd.getUTCDate();
  }
  var diff=Math.round((Date.UTC(adjY,adjM-1,adjD)-Date.UTC(1900,0,1))/86400000);
  return {k:((diff)%10+10)%10,s:((10+diff)%12+12)%12};
}

function getHour(h,dk){
  if(h<0) return null;
  var shi=Math.floor((h+1)/2)%12;
  var kan=(HKB[dk]+shi)%10;
  return {k:kan,s:shi};
}

function tsuhen(nk,zk){
  var n5=KG5[nk],z5=KG5[zk],ny=nk%2,zy=zk%2;
  var sg=(n5+1)%5,nko=KM[n5],zko=KM[z5];
  if(z5===n5)    return ny===zy?0:1;
  if(z5===sg)    return ny===zy?2:3;
  if(z5===nko)   return ny===zy?4:5;
  if(zko===n5)   return ny===zy?6:7;
  return ny===zy?8:9;
}

function juniUnsei(nk,si){
  var isYo=nk%2===0;
  var CSH_IN={1:6,3:9,5:9,7:0,9:3};
  var cs=isYo?CSH[nk]:CSH_IN[nk];
  return isYo?(si-cs+12)%12:(cs-si+12)%12;
}

function kubo(dk,ds){
  var n=0;
  for(var i=0;i<60;i++){if(i%10===dk&&i%12===ds){n=i;break;}}
  var jun=Math.floor(n/10);
  var kl=[[10,11],[8,9],[6,7],[4,5],[2,3],[0,1]];
  return kl[jun];
}

function getNeSokan(nk, ns){
  var num=0;
  for(var i=0;i<60;i++){if(i%10===nk&&i%12===ns){num=i;break;}}
  if(num<=4)  return 9;
  if(num<=19) return 8;
  if(num<=29) return 9;
  if(num<=49) return 8;
  return 9;
}

function getZokanFull(shi, pos, nesokan){
  if(shi===0) return nesokan;
  return (pos===2) ? YOKKI[shi] : HONKI[shi]; // 日柱のみ余気、年月時は本気
}

function calcGogyo(cols,nk,nesokan){

    if(!col) continue;
    c[JKG[col.k]]++;

function calcShiouLevel(cols,nk,zokanArr){
  var score=0;
  cols.forEach(function(col,i){
    if(!col)return;
    if(i!==2){
      var tk=tsuhen(nk,col.k);
      if(tk<=1||tk>=8) score+=1; else score-=1;
    }
    if(zokanArr&&zokanArr[i]>=0){
      var ts=tsuhen(nk,zokanArr[i]);
      if(ts<=1||ts>=8) score+=1; else score-=1;
    }
  });

  var geturei=false;
  if(cols[1]){
    var ju=juniUnsei(nk,cols[1].s);
    if(ju===4){score+=4;geturei=true;}
    else if(ju===3){score+=3;geturei=true;}
    else if(ju===2){score+=2;geturei=true;}
    else if(ju===0){score+=1;geturei=true;}
    else if(ju===1){score+=1;}
  }

  if(cols[2]&&zokanArr&&zokanArr[2]>=0){
    var nicST=tsuhen(nk,zokanArr[2]);
    if(nicST<=1) score+=1;
  }

  var level=(score>=0)?'身旺':'身弱';
  level += geturei ? ' (月令を得ています)' : ' (月令を得ていません)';
  return {level:level,score:score,geturei:geturei};
}

function calcDaiaun(y,m,d,h,mi,nenKan,gender,mc){
  var nenYo = nenKan%2===0;
  var forward = (gender==='男'&&nenYo)||(gender==='女'&&!nenYo);

  var birth=dateUTC(y,m,d,h,mi);
  var targetSetsu = forward ? getNextSetsuUTC(y,m,d,h,mi) : getPrevSetsuUTC(y,m,d,h,mi);
  var diffDays = Math.abs(targetSetsu-birth)/86400000;

  // 3日=1年、1日=4か月。小数日も一応月に反映。
  var totalMonths = Math.round(diffDays * 4);
  var startAgeYears = Math.floor(totalMonths / 12);
  var startAgeMonths = totalMonths % 12;

  var startAge = startAgeYears; // 表示用の年齢帯は年単位
  var results = [];
  for(var i=0;i<=9;i++){
    var ageFrom, ageTo, dk, ds;
    if(i===0){
      ageFrom = 0;
      ageTo = startAge;
      dk = mc.k;
      ds = mc.s;
    } else {
      ageFrom = startAge + (i-1)*10 + 1;
      ageTo = ageFrom + 9;
      if(forward){
        dk = ((mc.k+i)%10+10)%10;
        ds = ((mc.s+i)%12+12)%12;
      } else {
        dk = ((mc.k-i)%10+10)%10;
        ds = ((mc.s-i)%12+12)%12;
      }
    }
    results.push({ageFrom:ageFrom,ageTo:ageTo,k:dk,s:ds,isFirst:i===0});
  }
  return {startAge:startAge,startAgeYears:startAgeYears,startAgeMonths:startAgeMonths,list:results,forward:forward};
}

function getCurrentAge(y,m,d){
  var today=new Date();
  var age=today.getFullYear()-y;
  var bd=new Date(today.getFullYear(),m-1,d);
  if(today<bd) age--;
  return age;
}

  var sho=calcKanshiShou(cols,[nenZ,getZ,nicZ,jikZ!=null?jikZ:-1]);

  var zokanArr=[nenZ,getZ,nicZ,jikZ!=null?jikZ:-1];
  var ankouMap={0:5,5:0,1:6,6:1,2:7,7:2,3:8,8:3,4:9,9:4};
  var ankouNames=['年','月','日','時'];
  for(var ai=0;ai<4;ai++) for(var bi=ai+1;bi<4;bi++){
    if(zokanArr[ai]>=0&&zokanArr[bi]>=0&&ankouMap[zokanArr[ai]]===zokanArr[bi]){
      sho.kannkuu.push(ankouNames[ai]+'-'+ankouNames[bi]);
    }
  }

  var da=calcDaiaun(y,m,d,calcHour,calcMinute,nen.k,gender,get);
  var shiouResult=calcShiouLevel(cols,nk,zokanArr);
  var shiouText=shiouResult.level.replace(/^身旺/,'身旺 '+shiouResult.score).replace(/^身弱/,'身弱 '+shiouResult.score);
  var kisatsu=calcKisatsu(nen,get,nic,jik,y,m,d,gender);

  renderResult({
    name:name,y:y,m:m,d:d,h:h,minute:minute,gender:gender,
    nen:nen,get:get,nic:nic,jik:jik,
    nenZ:nenZ,getZ:getZ,nicZ:nicZ,jikZ:jikZ,
    nenKT:nenKT,getKT:getKT,jikKT:jikKT,
    nenST2:nenST2,getST2:getST2,nicST2:nicST2,jikST2:jikST2,
    nenJ:nenJ,getJ:getJ,nicJ:nicJ,jikJ:jikJ,
    kb:kb,gogyo:gogyo,sho:sho,da:da,shiouText:shiouText,nk:nk,kisatsu:kisatsu,
    currentAge:getCurrentAge(y,m,d)
  });
}
